Usage Guidelines

Import Components

// Atomic Design Components
import Container from '@/components/atoms/Container/Container.astro';
import Button from '@/components/atoms/Button/Button.astro';
import StatCard from '@/components/molecules/StatCard/StatCard.astro';
import FooterStatCard from '@/components/molecules/FooterStatCard/FooterStatCard.astro';
import Footer from '@/components/sections/Footer.astro';

// Design Tokens
import { semantic } from '@/tokens/semantic';

Design Tokens in CSS

/* Use CSS custom properties */
background: var(--hg-yellow-primary);
color: var(--hg-gray-dark);
padding: var(--hg-section-padding);

Component Usage

<!-- Atomic Design in Practice -->
<Container>
  <FooterStatCard number="97%" description="loan approval rate" />
  <Button variant="primary" size="lg" href="/contact">Get Started</Button>
</Container>

<!-- Astro Tokens -->
<style define:vars={{ brandPrimary: semantic.color.brand.primary }}>
  .footer { background: var(--brandPrimary); }
</style>
